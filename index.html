function generateTask() { var n = appState.num; if(!n) { alert('Выбери номер задания!'); return; } var taskData = TASKS_DB[n]; if(!taskData) { alert('Эта задача пока в разработке! 🚧'); return; } var topic = taskData.subTopics ? taskData.subTopics[appState.subIndex] : null; if(!topic || !topic.tasks || topic.tasks.length === 0) { alert('Подпункт пока в разработке! 🚧'); return; } var taskIdx = getNextTaskIndex(n, appState.subIndex); var variant = topic.tasks[taskIdx]; var params = variant.gen(); // 👇 ПРОВЕРКА НА ПОВТОРЕНИЕ ПАРАМЕТРОВ var maxAttempts = 10; var attempts = 0; while(appState.lastParams && JSON.stringify(appState.lastParams) === JSON.stringify(params) && attempts < maxAttempts) { params = variant.gen(); attempts++; } appState.lastParams = params; // 👆 КОНЕЦ ПРОВЕРКИ var svgHtml = variant.svg(params); var qTxt = variant.txt(params); var solR = variant.solRight(params); var solW = variant.solWrong(params); var corrAns = variant.ans(params); appState.currentSvg = svgHtml; document.getElementById('drawing-area').innerHTML = svgHtml; appState.params = { ans: corrAns, solRight: solR, solWrong: solW, num: n }; document.getElementById('menu-screen').classList.remove('active'); document.getElementById('task-screen').classList.add('active'); document.getElementById('lbl-type').textContent = taskData.title; document.getElementById('lbl-num').textContent = 'Задание №' + n; document.getElementById('q-text').textContent = qTxt; document.getElementById('btn-check').style.display = 'flex'; document.getElementById('btn-continue').style.display = 'none'; resetUI(); }
